PROG = run 							# исполняемый файл	
CC = gcc							# команда для компиляции
SRCS = main.c factorial.c			# исходные файлы
HDRS = factorial.h                  # заголовочные файлы
OBJS = $(SRCS:.c=.o)				# заменяет все исходные файлы на объектные

all: $(PROG)

$(PROG): $(OBJS)					# правило для сборки программы (есть ли объектные файлы?)
	$(CC) -o $(PROG) $(OBJS)		# если есть, то компилируем

clean:								# правило очистки объектных файлов
	rm -f $(OBJS)					# команда удаления объектных файлов

.c.o : $(HDRS) ; $(CC) -c $*.c  	# suffix rule
									# если HDRS изменится, то все объектные файлы будут пересобранны
									# команда $(CC) -c $*.c производит компиляцию без линковки
